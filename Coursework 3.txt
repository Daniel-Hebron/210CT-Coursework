s = (int(input('Input Number: ')))                      [1]
ans = 0                                                 [1]
if s >= 0:						[1]
    while ans*ans <= s:					[n]
        ans += 1					[n]
    if ans*ans != s:					[n]
        ans=ans -1					[n]
        print('The Highest Square Number is', ans)	[1]
    else: print('The Highest Square Number is', ans)	[1]
else: print(s,'is an error')				[1]
